//*************class.h
#include<string>
class list {
	std::string name;
	list *hand;
public:
	list(std::string str= "", list *h = 0) { name = str, hand = h;}
	void clean();
	list* findnum(int n);
	int findname(std::string str);
	void next(list *a);
	void del(int n);
	void showlist();
	void showlength();
	friend void insert(list *lh, list *exlh);
};




          // ************************class.cpp
#include"stdafx.h"
#include"class.h"
#include<iostream>
#include<string>

void insert(list * lh,list * exlh)
{
	lh->next(exlh->hand);
	exlh->next(lh);
}

void list::del(int n)
{
	list *b = this,*t;
	for (int a = 0; a < n; a++)
	{
		t = b;
		b = b->hand;
		if (b == 0)
		{
			std::cout << "wrong";
			break;
		}
	}
	if (b != 0)
	{
		t->next(b->hand);
		delete(b);
	}
}

void list::next(list *a)
{
	hand = a;
}

void list::clean()
{
	list *a=this->hand,*b;
	while (a != 0)
	{
		b = a;
		a = a->hand;
		delete(b);
	}
	this->hand = 0;
}

int list::findname(std::string str)
{
	int n = 0;
	list *b = this;
	while (b != 0)
	{
		if (str == b->name)
			break;
		b = b->hand;
		n++;
	}
	if (b == 0)
	{
		std::cout << "列表中没有这个名字";
		return 0;
	}
	else
		return n;
}

list* list::findnum(int n)
{
	list *b=this;
	for (int a = 0; a < n; a++)
	{
		b = b->hand;
		if (b == 0)
		{
			std::cout << "wrong";
			break;
		}
	}
	if (b != 0)
		return b;
	else
		return 0;
}
